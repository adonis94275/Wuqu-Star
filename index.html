const MarketRegime = { BULL_TREND: 'BULL_TREND', // 適合進攻 SIDEWAYS: 'SIDEWAYS', // 適合低買高賣 BEAR_MARKET: 'BEAR_MARKET' // 全面防守/禁止多單 }; function executeASQEv5Engine(stockData, marketData) { // 1. 辨識市場階段 (Market Regime Engine) const regime = determineMarketRegime(marketData); // 2. 基礎多因子計分 (升級版品質與 RS Score) let baseScore = 0; baseScore += evaluateMarketEnvironment(marketData); baseScore += evaluateQualityScore(stockData); // 升級：連兩季EPS、毛利、FCF baseScore += evaluateRelativeStrength(stockData); // 升級：20/60/120日加權RS Ranking baseScore += evaluateAnchoredVWAP(stockData); // 升級：AVWAP30/60替代傳統POC // 3. 雙層否決機制 (Hard / Soft Veto) const vetoResults = checkVetoConditions(stockData, marketData); let finalScore = baseScore; let systemStatus = "NORMAL"; let actionAdvice = ""; if (vetoResults.hasHardVeto) { finalScore = Math.min(finalScore, 40); // 強制鎖死轉弱區 systemStatus = "HARD_VETO"; actionAdvice = vetoResults.hardVetoReason + " ｜ 立即全數清倉！"; } else if (vetoResults.hasSoftVeto) { finalScore -= vetoResults.softVetoPenalty; // 扣分制（10~20分） systemStatus = "SOFT_VETO"; actionAdvice = vetoResults.softVetoReason + " ｜ 建議逢高減碼。"; } // 4. 自適應門檻與動態部位管理 (Position Sizing Engine) const allocation = calculateDynamicPosition(finalScore, regime, stockData.atrPercentage); return { regime: regime, score: finalScore, status: systemStatus, positionSize: allocation.size, advice: actionAdvice || allocation.advice, stopLoss: allocation.stopLoss, takeProfit: allocation.takeProfit }; } // 模組①：市場階段自適應 function determineMarketRegime(market) { const isAboveEMA200 = market.close > market.ema200; const isTrending = market.adx > 25; if (isAboveEMA200 && isTrending) return MarketRegime.BULL_TREND; if (!isAboveEMA200) return MarketRegime.BEAR_MARKET; return MarketRegime.SIDEWAYS; } // 模組②：動態部位管理 (結合市場階段與 Kelly Criterion 概念) function calculateDynamicPosition(score, regime, atrPct) { let baseSize = 0; let advice = ""; // 根據 Regime 與 分數 自適應調整進場門檻 switch(regime) { case MarketRegime.BULL_TREND: if (score >= 85) { baseSize = 0.12; advice = "SS 主升段強攻"; } else if (score >= 70) { baseSize = 0.08; advice = "S 潛力股順勢推進"; } else { baseSize = 0.00; advice = "空手觀望"; } break; case MarketRegime.SIDEWAYS: if (score >= 80) { baseSize = 0.06; advice = "S 頂級標的高檔震盪，嚴格控盤"; } else { baseSize = 0.00; advice = "區間不明，空手觀望"; } break; case MarketRegime.BEAR_MARKET: if (score >= 90) { baseSize = 0.03; advice = "極端防禦白名單，極小額試單"; } else { baseSize = 0.00; advice = "空頭市場，嚴禁建立多單"; } break; } // 波動率調節 (ATR% 過高則縮減部位，保護帳戶最大回撤) if (atrPct > 0.05) baseSize *= 0.8; return { size: (baseSize * 100).toFixed(1) + "%", advice: advice }; } // 模組③：雙層否決機制 function checkVetoConditions(stock, market) { let hasHardVeto = false; let hardVetoReason = ""; let hasSoftVeto = false; let softVetoPenalty = 0; let softVetoReason = ""; // Hard Veto: 結構性破壞 if (stock.close < stock.ema60) { hasHardVeto = true; hardVetoReason = "🚨 跌破生命線 EMA60"; } else if (stock.volume > 2.0 * stock.vol20MA && stock.isBlackK) { hasHardVeto = true; hardVetoReason = "🚨 高檔爆量長黑（主力無情出貨）"; } // Soft Veto: 較小風險扣分制 if (stock.instThreeDaysSell) { hasSoftVeto = true; softVetoPenalty += 15; softVetoReason += "⚠️ 法人短線調節 "; } if (stock.upperShadowPercent > 0.04) { hasSoftVeto = true; softVetoPenalty += 10; softVetoReason += "⚠️ 上方留下長上影線壓力 "; } return { hasHardVeto, hardVetoReason, hasSoftVeto, softVetoPenalty, softVetoReason }; }
